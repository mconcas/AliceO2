35 struct Triggers {
36  enum { bitA,
37  bitAIn = 1,
38  bitC = 1, // alias of bitAIn - needed to be compatible with common-for-FIT raw reader
39  bitAOut = 2,
40  bitVertex = 2, // alias of bitAOut - needed to be compatible with common-for-FIT raw reader
41  bitCen,
42  bitSCen,
43  bitLaser };
44  uint8_t triggersignals = 0; // V0 trigger signals
45  int8_t nChanA = 0; // number of fired channels A side
46  int8_t nChanC = 0; // TODO: MS: unused in FV0
47  int32_t amplA = -5000; // sum amplitude A side
48  int32_t amplC = -5000; // TODO: MS: unused in FV0
49  int16_t timeA = -5000; // average time A side
50  int16_t timeC = -5000; // TODO: MS: unused in FV0
51  Triggers() = default;
52  Triggers(uint8_t signals, int8_t chanA, int32_t aamplA, int16_t atimeA)
53  {
54  triggersignals = signals;
55  nChanA = chanA;
56  amplA = aamplA;
57  timeA = atimeA;
58  }
59  bool getOrA() const { return (triggersignals & (1 << bitA)) != 0; }
60  bool getOrAIn() const { return (triggersignals & (1 << bitAIn)) != 0; }
61  bool getOrAOut() const { return (triggersignals & (1 << bitAOut)) != 0; }
62  bool getCen() const { return (triggersignals & (1 << bitCen)) != 0; }
63  bool getSCen() const { return (triggersignals & (1 << bitSCen)) != 0; }
64  bool getLaserBit() const { return (triggersignals & (1 << bitLaser)) != 0; }
65 
66  // TODO: MS: temporary aliases to keep DigitBlockFIT.h working (treat FV0 as FT0/FDD)
67  bool getOrC() const { return getOrAIn(); }
68  bool getVertex() const { return getOrAOut(); }
69 
70  void setTriggers(Bool_t isA, Bool_t isAIn, Bool_t isAOut, Bool_t isCnt, Bool_t isSCnt, int8_t chanA, int32_t aamplA,
71  int16_t atimeA, Bool_t isLaser = kFALSE)
72  {
73  triggersignals = (isA << bitA) | (isAIn << bitAIn) | (isAOut << bitAOut) | (isCnt << bitCen) | (isSCnt << bitSCen) | (isLaser << bitLaser);
74  nChanA = chanA;
75  amplA = aamplA;
76  timeA = atimeA;
77  }
78  void cleanTriggers()
79  {
80  triggersignals = 0;
81  nChanA = 0;
82  amplA = -5000;
83  timeA = -5000;
84  }
85  bool operator==(Triggers const& other) const
86  {
87  return std::tie(triggersignals, nChanA, amplA, timeA) ==
88  std::tie(other.triggersignals, other.nChanA, other.amplA, other.timeA);
89  }
90  void printLog() const;
91  ClassDefNV(Triggers, 2);
92 };
